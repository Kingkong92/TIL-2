```python
# 3

def json(string) :
    middle_start = 0
    middle_end = 0
    large_start = 0
    large_end = 0
    
    for s in string :
        if s == "{" :
            middle_start += 1
        elif s == "}" :
            middle_end += 1
        elif s == "[" :
            large_start += 1
        elif s == "]" :
            large_end += 1
        elif (large_start != large_end) and (s == ":") :
            break
    if (middle_start == middle_end) and (large_start == large_end) :
        print("Valid JSON")
    else :
        print("Parse error!")
        
string = '{"level":3, "attr":{"power":120, "name":"hero"}, "friendIds":[12, 23, 34, 23]}'
json(string)
# > Valid JSON

string = '[12, 23, 34, "23"]'
json(string)
# > Valid JSON

string = '{"level":3, "attr":{"power":120, "name":"hero"}, "friendIds":[12, 23, 34, 23}'
json(string)
# > Parse error!

string = '["level":3, 23, 34, "23"]'

# > Parse error!
```
